import { StatusBar } from "expo-status-bar";
import { StyleSheet, Text, SafeAreaView, Platform } from "react-native";
import React, { useState } from "react";
import Home from "./src/Home";

export default function App() {
  const [pagina, setPagina] = useState(0);

  const handleMudancaPagina = (codPagina) => {
    setPagina(codPagina);
  };

  return (
    <SafeAreaView style={styles.container}>
      <Text>APP Main {pagina}</Text>
      {pagina < 10 ? ( <Home pagina={pagina} onMudancaPagina={handleMudancaPagina} />):<></>}
    </SafeAreaView>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: "white",
    paddingTop: Platform.OS === "android" ? StatusBar.currentHeight : 0,
  },
});


import React from "react";
import { Button, ScrollView } from "react-native";

export default function Home({ pagina, onMudancaPagina }) {
  // Ajuste nas props recebidas
  const retornarPagina = () => {
    onMudancaPagina(pagina + 1);
  };

  return (
    <ScrollView>
      <Button
        title="Botão"
        onPress={retornarPagina} // Definição correta da função onPress
      />
    </ScrollView>
  );
}
